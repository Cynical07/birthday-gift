  <script>

    const messages = [

      "I could’ve just sent you a playlist...",

      "But that wouldn’t show how much you mean to me.",

      "So I made this for you instead.",

      "Every song here reminded me of you.",

      "Happy Birthday, love."

    ];


    let messageIndex = 0;

    const msgBox = document.getElementById('message-box');


    function openCard() {

      document.getElementById('card-scene').style.display = 'none';

      document.getElementById('message-scene').style.display = 'flex';

      document.getElementById('bg-music').play();

      showNextMessage();

    }


    function showNextMessage() {

      if (messageIndex >= messages.length) {

        document.getElementById('message-scene').style.display = 'none';

        document.getElementById('gift-scene').style.display = 'flex';

        document.getElementById('gift-box').style.display = 'block';

        return;

      }

      msgBox.innerText = messages[messageIndex];

      msgBox.style.opacity = 1;

      msgBox.style.display = 'block';

      setTimeout(() => {

        msgBox.style.opacity = 0;

        setTimeout(() => {

          messageIndex++;

          showNextMessage();

        }, 2000);

      }, 2500);

    }


    function showPlaylist() {

      document.getElementById('gift-scene').style.display = 'none';

      document.getElementById('playlist-section').style.display = 'block';

    }

  </script>
